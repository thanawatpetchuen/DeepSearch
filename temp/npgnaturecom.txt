        (function() {
            var forEach = function (array, callback, scope) {
                for (var i = 0; i < array.length; i++) {
                callback.call(scope, i, array[i])
                }
            }

            function addCheckedClassToParentIfInputVisible(checkedField) {
                checkedField.parentNode.classList.add(
                    checkedField.offsetParent
                        ? 'checked'
                        : 'checked-invisible'
                )
            }

            function addCheckedClassToParent(checkedField) {
                checkedField.parentNode.classList.add('checked')
            }

            function passOriginalIndexToFormSubmitForTracking(form, checkedField) {
                form.querySelector('button').setAttribute(
                    'data-track-product',
                    checkedField.getAttribute('data-original-index')
                )
            }

            function reflectFormState(form) {
                var checkedField = form.querySelector(':checked')

                //addCheckedClassToParentIfInputVisible(checkedField)
                addCheckedClassToParent(checkedField)
                passOriginalIndexToFormSubmitForTracking(form, checkedField)
            }

            function radioChangeHandler() {
                forEach(this.form.querySelectorAll('.input-field'), function(index, inputField) {
                    inputField.classList.remove('checked')
                })

                reflectFormState(this.form)
            }

            function reflectFormStates() {
                forEach(document.querySelectorAll('[data-reflect-state]'), function(index, form) {
                    reflectFormState(form)
                })
            }

            function detailToggle(currentDetail) {
                return function () {
                    var openDetail = document.querySelector('[data-reflect-state] details[open]')
                    if (openDetail && openDetail !== currentDetail) openDetail.removeAttribute('open')
                }
            }

            function init() {
                var details = document.querySelectorAll('[data-reflect-state] details')
                var accessOptionDivs = document.querySelectorAll('.input-field')
                var radioInputFieldsRadio = document.querySelectorAll('.input-field input')

                forEach(details, function (index, detail) {
                    detail.addEventListener('click', detailToggle(detail), false)
                })

                forEach(radioInputFieldsRadio, function (index, inputField) {
                    inputField.addEventListener('change', radioChangeHandler, false)
                })

                forEach(accessOptionDivs, function (index, accessOptionDiv) {
                    accessOptionDiv.addEventListener('click', function () {
                        this.querySelector('input').click()
                    })
                })

                reflectFormStates()
            }

            init()

        })()
